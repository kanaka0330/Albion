  <script>
    const size = 5;
    let cardNumbers = [];   // 自分のカード
    const drawnNumbers = [];

    const card = document.getElementById("bingo-card");
    const drawnDiv = document.getElementById("drawn-numbers");
    const nameInput = document.getElementById("player-name");
    const participantList = document.getElementById("participant-list");

    // 🔹 ビンゴカードを生成
    function generateCard() {
      const nums = [];
      const used = new Set();
      for (let i = 0; i < size * size; i++) {
        let num;
        do {
          num = Math.floor(Math.random() * 75) + 1;
        } while (used.has(num));
        used.add(num);
        nums[i] = num;
      }
      nums[Math.floor(size * size / 2)] = "FREE";
      return nums;
    }

    // 🔹 ビンゴカードを表示
    function displayCard(numbers) {
      card.innerHTML = "";
      numbers.forEach(n => {
        const cell = document.createElement("div");
        cell.className = "cell";
        cell.textContent = n;
        if (n === "FREE") cell.classList.add("marked");
        card.appendChild(cell);
      });
    }

    // 🔹 プレイヤーを登録
    function registerPlayer() {
      const name = nameInput.value.trim();
      if (!name) return alert("名前を入力してください！");

      const ref = database.ref("participants/" + name);

      ref.once("value").then((snap) => {
        if (snap.exists()) {
          // 既存カードがある → それを利用
          const data = snap.val();
          cardNumbers = data.card;
          displayCard(cardNumbers);
          alert(`${name} さんは既存カードを読み込みました！`);
        } else {
          // 新規参加 → カード生成して保存
          cardNumbers = generateCard();
          ref.set({
            card: cardNumbers,
            status: "参加中"
          });
          displayCard(cardNumbers);
          alert(`${name} さんのカードを新規作成しました！`);
        }
      });

      nameInput.value = "";
    }

    // 🔹 参加者リストを同期表示
    database.ref("participants").on("value", (snap) => {
      participantList.innerHTML = "";
      const data = snap.val();
      if (!data) return;
      Object.keys(data).forEach(name => {
        const li = document.createElement("li");
        li.textContent = `${name}（${data[name].status}）`;
        participantList.appendChild(li);
      });
    });

    // 🔹 番号をひく（主催者用）
    function drawNumber() {
      if (drawnNumbers.length >= 75) return;
      let num;
      do {
        num = Math.floor(Math.random() * 75) + 1;
      } while (drawnNumbers.includes(num));

      drawnNumbers.push(num);
      highlightNumber(num);
      updateDrawnNumbers();

      // Firebaseに保存（順番保持）
      database.ref("drawnNumbers").set(drawnNumbers);
    }

    // 🔹 番号を強調表示
    function highlightNumber(n) {
      document.querySelectorAll(".cell").forEach(c => {
        if (parseInt(c.textContent) === n) {
          c.classList.add("marked");
        }
      });
    }

    // 🔹 番号の表示更新
    function updateDrawnNumbers() {
      drawnDiv.innerHTML = `<strong>出た番号：</strong> ${drawnNumbers.join("・")}`;
    }

    // 🔹 Firebaseから抽選結果を反映
    database.ref("drawnNumbers").on("value", (snapshot) => {
      const data = snapshot.val();
      if (data) {
        drawnNumbers.length = 0;
        drawnNumbers.push(...data);
        updateDrawnNumbers();
        data.forEach(n => highlightNumber(n));
      }
    });
  </script>
